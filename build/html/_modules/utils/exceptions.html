import sys



[docs]class BaseError(RuntimeError):
    """ Error base class

    """
    def __init__(self, arg, err_id=None):
        self.arg = arg
        self.err_id = err_id

    def __str__(self):
        if self.err_id is None:
            return self.arg
        else:
            return "error=%d, %s" % (self.err_id, self.arg)
